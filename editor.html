function doGet(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName('상가DB') || ss.getSheets()[0];
  const data = sheet.getDataRange().getValues();
  const headers = data[0].map(String);

  const buildingCol = headers.findIndex(h =>
    ['상가명','상가','building'].includes(h.trim()) || h.includes('상가')
  );

  const mode = e.parameter.mode;
  let result;

  if (mode === 'list') {
    const names = [...new Set(
      data.slice(1)
        .map(r => buildingCol >= 0 ? String(r[buildingCol] || '').trim() : '')
        .filter(v => v)
    )].sort();
    result = { buildings: names };
  } else {
    const building = e.parameter.building || '';
    const rows = data.slice(1)
      .filter(r => !building || (buildingCol >= 0 && String(r[buildingCol]).trim() === building))
      .map(r => {
        const obj = {};
        headers.forEach((h, i) => { obj[h] = r[i] != null ? String(r[i]) : ''; });
        return obj;
      });
    result = { headers, rows };
  }

  const cb = e.parameter.callback || '';
  const json = JSON.stringify(result);
  const out = cb ? cb + '(' + json + ')' : json;
  const mime = cb ? ContentService.MimeType.JAVASCRIPT : ContentService.MimeType.JSON;
  return ContentService.createTextOutput(out).setMimeType(mime);
}
